DROP TABLE UTILISATEUR, FORMATION, CONFERENCE ;

CREATE TABLE UTILISATEUR (
	ID INTEGER,
	PSEUDO VARCHAR(10), /* identifiant de connexion */
	MDP VARCHAR(30),
	STATUT VARCHAR(10), /* admin,citoyen ou visiteur */
	NOM VARCHAR(20),
	PRENOM VARCHAR(20),
	AGE INTEGER,
	SITUATION_PRO VARCHAR(20),
	ETUDES VARCHAR(20),
	/* Autres infos */ 
	CONSTRAINT user_PK PRIMARY KEY(ID)
    );
CREATE TABLE FORMATION (
	
	TITRE	VARCHAR(40),
	THEME VARCHAR(15),
	DATEF DATE,
	FORMATEUR VARCHAR(20),
	CONSTRAINT formation_PK PRIMARY KEY(TITRE,DATEF),
	);

CREATE TABLE CONFERENCE(
	TITRE VARCHAR(20),
	DATEC DATE,
	THEME VARCHAR(15),
	/* Participants/Votants , r√©sultat ?*/
	CONSTRAINT conference_PK PRIMARY KEY(TITRE,DATEC),
